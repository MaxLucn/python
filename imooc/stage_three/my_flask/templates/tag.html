if 的使用

{% if var is none -%}

var 的值是 none

{% else -%}

var 的值不是 none

{% endif -%} {% if a is defined -%}

a 是被定义好的

{% else -%}

a 是没有定义好的

{% endif -%}

for 循环的使用

{% for item in my_list -%}

用户名：{{ item.username }}, 年龄: {{ item.age }}，地址: {{ item.address }}

{% else -%}

用户信息为空

{% endfor -%}

for 循环 dict

{% for user in my_list -%}

第 {{ loop.index }} 个用户：
{% for key, value in user.items() -%} {{ key }}: {{value}} {% endfor -%}

{% else -%}

用户信息为空

{% endfor -%}

for 循环 dict-break

{% for user in my_list -%}

第 {{ loop.index }} 个用户：
{% for key, value in user.items() -%} {% if loop.index > 2 -%} {% break -%} {% endif -%} {{ key }}: {{value}} {% endfor -%}

{% else -%}

用户信息为空

{% endfor -%}

赋值的使用

{% with %} {% set value = 42 %} {{ value }} {% endwith %}

模版语法的转义

{{ '{{' }}
{% raw %} {% with %} {% set value = 42 %} {{ value }} {% endwith %} {% endraw %} {# 这是一个模版注释，在浏览器中查看源码时不会显示 #}
